class MemoryManager:

        self.scratchpad_prefix = "maxey:scratchpad:"
        self.persistent_prefix = "maxey:persistent:"
    
    async def store_scratchpad(self, agent_id: str, data: Dict[str, Any], ttl: int = 3600):
        """Store data in scratchpad (Redis) with TTL"""
        key = f"{self.scratchpad_prefix}{agent_id}:{uuid.uuid4().hex[:8]}"
        await self.redis.setex(key, ttl, json.dumps(data))
        return key
    
    async def move_to_persistent(self, scratchpad_key: str, agent_id: str):
        """Move data from scratchpad to persistent memory (Neo4J)"""
        data = await self.redis.get(scratchpad_key)
        if data:
            persistent_key = f"{self.persistent_prefix}{agent_id}:{uuid.uuid4().hex[:8]}"
            # Store in Redis for quick access and Neo4J for durability
            await self.redis.set(persistent_key, data)

                )

class ScratchpadManager:

    async def store(self, entry: MemoryEntry) -> str:
        """Store entry in scratchpad with TTL"""
        key = f"{self.prefix}{entry.source_agent}:{entry.memory_id}"
        
        # Serialize entry
        data = {
            "memory_id": entry.memory_id,
            "content": entry.content,
            "memory_type": entry.memory_type.value,
            "source_agent": entry.source_agent,
            "importance_score": entry.importance_score,

            "last_accessed": entry.last_accessed.isoformat(),
            "tags": entry.tags,
            "metadata": entry.metadata
        }
        
        ttl = entry.ttl_seconds or self.default_ttl
        
        try:
            self.redis.setex(key, ttl, json.dumps(data))
            logger.info(f"Stored scratchpad entry {entry.memory_id} with TTL {ttl}s")
            return key
        except Exception as e:
            logger.error(f"Failed to store scratchpad entry: {e}")

            if not data:
                return None
            
            parsed_data = json.loads(data)
            
            # Update access count and timestamp
            parsed_data["access_count"] += 1
            parsed_data["last_accessed"] = datetime.now().isoformat()

class BaseTrigger(ABC):
    """Base class for all spawning triggers"""
    
    def __init__(self, trigger_id: str, trigger_type: TriggerType, threshold: float):
        self.trigger_id = trigger_id
        self.trigger_type = trigger_type
        self.threshold = threshold
        self.metrics = TriggerMetrics(trigger_id, trigger_type, threshold, 0.0)
        self.is_enabled = True
        self.cooldown_seconds = 60  # Prevent rapid successive triggers
        self.last_activation = None
    
    @abstractmethod
    async def evaluate(self, context: SpawningContext) -> Tuple[bool, float]:
        """Evaluate if trigger should fire"""
        pass
    
    @abstractmethod
    def get_recommended_scw_types(self) -> List[str]:
        """Get recommended SCW types for this trigger"""
        pass
    
    async def is_ready_to_fire(self) -> bool:
        """Check if trigger is ready to fire (cooldown check)"""
        if not self.is_enabled:
            return False
        
        if self.last_activation is None:
            return True
        
        time_since_last = (datetime.now() - self.last_activation).total_seconds()
        return time_since_last >= self.cooldown_seconds
    
    async def record_activation(self, success: bool, response_time: float):
        """Record trigger activation and outcome"""
        self.metrics.activation_count += 1
        self.last_activation = datetime.now()
        self.metrics.last_triggered = self.last_activation
        
        # Update success rate with exponential moving average
        if self.metrics.activation_count == 1:
            self.metrics.success_rate = 1.0 if success else 0.0
        else:
            alpha = 0.1  # Smoothing factor
            self.metrics.success_rate = (
                alpha * (1.0 if success else 0.0) + 
                (1 - alpha) * self.metrics.success_rate
            )
        
        # Update average response time
        if self.metrics.activation_count == 1:
            self.metrics.average_response_time = response_time
        else:
            alpha = 0.2
            self.metrics.average_response_time = (
                alpha * response_time + 
                (1 - alpha) * self.metrics.average_response_time
            )

        scratchpad_utilization = context.memory_usage.get("scratchpad_utilization", 0.0)
        
        # Calculate pressure score

        utilization_pressure = scratchpad_utilization

        should_trigger = pressure_score > 1.0

class TaskComplexityTrigger(BaseTrigger):

    async def _calculate_complexity(self, task: Dict[str, Any]) -> float:
        """Calculate task complexity score"""
        complexity_score = 0.0
        
        # Text length factor
        task_text = str(task)
        text_complexity = min(len(task_text) / 5000, 0.3)
        complexity_score += text_complexity
        
        # Keyword complexity
        complex_keywords = [
            'multiagent', 'orchestration', 'distributed', 'coordination',
            'optimization', 'architecture', 'integration', 'synthesis',
            'analysis', 'generation', 'transformation', 'pipeline'
        ]
        
        text_lower = task_text.lower()
        keyword_matches = sum(1 for keyword in complex_keywords if keyword in text_lower)
        keyword_complexity = min(keyword_matches * 0.1, 0.4)
        complexity_score += keyword_complexity
        
        # Task type complexity
        task_type = task.get('type', '').lower()
        type_complexity = {
            'multiagent': 0.9,
            'system_design': 0.8,
            'architecture': 0.7,
            'integration': 0.6,
            'analysis': 0.5,
            'generation': 0.4,
            'simple': 0.2
        }
        
        for task_pattern, complexity in type_complexity.items():
            if task_pattern in task_type:
                complexity_score += complexity * 0.3
                break
        
        # Data processing complexity
        if 'data' in task:
            data_size = len(str(task.get('data', '')))
            data_complexity = min(data_size / 10000, 0.2)
            complexity_score += data_complexity
        
        # Requirements complexity
        requirements = task.get('requirements', [])
        if isinstance(requirements, list):
            req_complexity = min(len(requirements) * 0.05, 0.2)
            complexity_score += req_complexity
        
        return min(complexity_score, 1.0)

class MCPClientManager:
    """Manages connections to other MCP servers as a client"""
    
    def __init__(self):
        self.connections: Dict[str, MCPConnection] = {}
        self.capabilities = [
            "agent_delegation",
            "scw_sharing", 
            "memory_sync",
            "file_processing",
            "a2a_communication"
        ]
    
    async def discover_servers(self) -> List[Dict[str, Any]]:
        """Auto-discover MCP servers on the network"""
        discovered = []
        
        # Common MCP server ports and endpoints
        common_endpoints = [
            "http://localhost:8001/mcp",
            "http://localhost:8002/mcp", 
            "http://localhost:9000/mcp",
            "http://127.0.0.1:8001/mcp"
        ]
        
        for endpoint in common_endpoints:
            try:
                async with aiohttp.ClientSession(timeout=aiohttp.ClientTimeout(total=5)) as session:
                    async with session.get(f"{endpoint}/capabilities") as response:
                        if response.status == 200:
                            capabilities = await response.json()
                            discovered.append({
                                "url": endpoint,
                                "capabilities": capabilities,
                                "status": "available"
                            })
                            logger.info(f"Discovered MCP server at {endpoint}")
            except Exception as e:
                logger.debug(f"No MCP server at {endpoint}: {e}")
        
        return discovered
    
    async def connect_to_server(self, url: str, name: str) -> bool:
        """Connect to an MCP server as a client"""
        try:
            connection_id = str(uuid.uuid4())
            
            # Test connection
            async with aiohttp.ClientSession() as session:
                async with session.get(f"{url}/capabilities") as response:
                    if response.status == 200:
                        capabilities = await response.json()
                        
                        connection = MCPConnection(
                            connection_id=connection_id,
                            name=name,
                            url=url,
                            capabilities=capabilities.get("capabilities", []),
                            status="connected"
                        )
                        
                        self.connections[connection_id] = connection
                        logger.info(f"Connected to MCP server {name} at {url}")
                        return True
            
            return False
            
        except Exception as e:
            logger.error(f"Failed to connect to MCP server {url}: {e}")
            return False
    
    async def send_request(self, connection_id: str, method: str, params: Dict[str, Any]) -> Dict[str, Any]:
        """Send request to connected MCP server"""
        if connection_id not in self.connections:
            raise ValueError(f"No connection found: {connection_id}")
        
        connection = self.connections[connection_id]
        
        message = MCPMessage(
            id=str(uuid.uuid4()),
            type=MCPMessageType.REQUEST,
            method=method,
            params=params
        )
        
        try:
            async with aiohttp.ClientSession() as session:
                async with session.post(
                    f"{connection.url}/request",
                    json={
                        "id": message.id,
                        "method": method,
                        "params": params
                    }
                ) as response:
                    if response.status == 200:
                        return await response.json()
                    else:
                        raise Exception(f"Request failed: {response.status}")
                        
        except Exception as e:
            logger.error(f"Request to {connection.name} failed: {e}")
            raise
    
    async def delegate_to_agent(self, connection_id: str, agent_id: str, task: Dict[str, Any]) -> Dict[str, Any]:
        """Delegate task to agent on connected MCP server"""
        return await self.send_request(connection_id, "agent_execute", {
            "agent_id": agent_id,
            "task": task
        })
    
    async def request_scw(self, connection_id: str, scw_type: str, config: Dict[str, Any]) -> Dict[str, Any]:
        """Request SCW creation on connected MCP server"""
        return await self.send_request(connection_id, "scw_create", {
            "scw_type": scw_type,
            "config": config
        })

## 🌐 MCP Server Discovery

Maxey0 automatically discovers other MCP servers on the network:

```python
# Auto-discovery process
async def discover_mcp_servers():
    """Find other MCP servers automatically"""
    
    # Scan common ports
    common_ports = [8001, 8002, 8003, 9000, 9001]
    discovered = []
    
    for port in common_ports:
        try:
            response = await aiohttp.get(f"http://localhost:{port}/mcp/capabilities")
            if response.status == 200:
                capabilities = await response.json()
                discovered.append({
                    "url": f"http://localhost:{port}",
                    "name": capabilities.get("name", f"Server_{port}"),
                    "capabilities": capabilities.get("capabilities", [])
                })
        except:
            continue
    
    return discovered
```
